■現在時刻の取得 (日本の場合は日本標準時)
    
        new Date()
    
    const d = new Date();
    
    ■UTC（協定世界時）
    
        d.getTime()
    
    ■月日に関する取得
    
        ▶︎ 年
        d.getFullYear

        ▶︎ 月
        d.getMonth

        ▶︎ 日
        d.getDate

        ▶︎ 曜日（0~6）(日曜日始まり)
        d.getDay

        ▶︎ 時間
        d.getHours

        ▶︎ 分
        d.getMinutes

        ▶︎ 秒
        d.getSeconds

        ▶︎ ミリ秒
        d.getMilliseconds
